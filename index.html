  const uniqueInOrder = (iterable) =>  {
    let result = [];

    if (typeof iterable === 'string') {
      iterable.split('');
    }

    for (let i = 0; i < iterable.length; i++) {
      if (iterable[i] !== iterable[i + 1]) {
        result.push(iterable[i]);
      }
    }
    return result;
  }
